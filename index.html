# sample: compute revenue growth
import pandas as pd

df = pd.read_csv('quarterly.csv')
prev = df[df['quarter']=='Q3']['revenue'].sum()
curr = df[df['quarter']=='Q4']['revenue'].sum()
growth = (curr - prev) / prev
print(f"Revenue growth: {growth:.2%}")
